解决在高德地图中循环加载marker并添加多个text标记文本，解决只显示最后一个标记文本的问题

Oct 14, 2020 About 7 mins

#高德地图#js

场景： 地图运动轨迹中有多个marker，不同marker使用不同点标记进行注释，该点触发的事件需要使用文本显示出来（最后是需要移入时显示，移开时隐藏）

难点： 高德地图的setlabel不能实现显示和隐藏；改用文本标记

问题： 在高德地图中循环加载marker并添加多个text标记文本，解决只显示最后一个标记文本的问题

解决方法： 使用闭包，注意this 的指向。

    for(let i = 0 ; i < this.alarmlist.length; i++){
        //解决循环加载marker并添加多个text标记文本，解决只显示最后一个标记文本的问题

        let _this = this;

        (function(i){

            let lat = _this.alarmlist[i].lat;

            let lon = _this.alarmlist[i].lon;



            marker = new AMap.Marker({
                icon: "./images/alarmMarker.png",

                position: [lon,lat],

                // offset: new AMap.Pixel(-13, -30)

                offset: new AMap.Pixel(-26, -13),

            });




            let alarmtime = _this.alarmlist[i].alarmtime;

            let subalarmtypestr = _this.alarmlist[i].subalarmtypestr;



            

            // 自定义点标记内容

            var markerContent =  "时间:" + alarmtime + "类型:" + subalarmtypestr +"<br/>" ;

                

            // marker.setLabel({
            //     offset: new AMap.Pixel(20, 20),  //设置文本标注偏移量

            //     content: "<div class='info' style='color: #fff'>"+ markerContent +"</div>", //设置文本标注内容

            //     direction: 'right' //设置文本标注方位

            // });



            // 创建纯文本标记

            let text = new AMap.Text({
                text: markerContent,

                anchor:'center', // 设置文本标记锚点

                draggable:true,

                cursor:'pointer',

                style:{
                    'margin-bottom': '1rem',

                    'background-color': 'rgba(0, 255, 255, 0.5)',

                    'text-align': 'center',

                    'font-size': '20px',

                    'color': '#fff',

                },

                position: [lon,lat],

            });

        

            AMap.event.addListener(marker, 'mouseover', function(e) {
            // console.log("监听事件mouseover",marker);

            // console.log("markerContent",markerContent);

            text.setMap(_this.Map);    

            })

            AMap.event.addListener(marker, 'mouseout', function(e) {
            console.log("监听事件mouseout");

            text.setMap(null);

            })

        

            _this.Map.add(marker);

        })(i);

    }
